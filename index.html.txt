            function createEffect(pos, color, size, decayMult) {
                const mesh = new THREE.Mesh(new THREE.SphereGeometry(size, 4, 4), new THREE.MeshBasicMaterial({ color: color, transparent: true, opacity: 0.4 }));
                mesh.position.copy(pos).add(new THREE.Vector3((Math.random()-0.5)*1.5, 0.1, (Math.random()-0.5)*1.5));
                scene.add(mesh);
                particles.push({ mesh: mesh, life: 1.0, decay: 0.025 * decayMult, grow: 1.03 });

            function createBackfire(v) {
                for(let i=0; i<5; i++) {
                    const p = v.mesh.position.clone();
                    const offset = new THREE.Vector3(0, 0.4, -1.8).applyAxisAngle(new THREE.Vector3(0, 1, 0), v.carAngle);
                    createEffect(p.add(offset), 0xffaa00, 0.4, 1.5);
                }
            }

            function createSkid(pos, angle) {

                scene.add(mesh);
                skids.push({ mesh: mesh, life: 1.0 });

            const gearCfg = {
                "-1": { acc: -0.004, max: -0.15 },
                "0": { acc: 0, max: 0 },
                "1": { acc: 0.007, max: 0.18 },
                "2": { acc: 0.006, max: 0.38 },
                "3": { acc: 0.005, max: 0.55 },

            };

            function animate() {
                requestAnimationFrame(animate);

                vehicles.forEach((v, idx) => {
                    if (v.isPlayer) {
                        let acc = 0;
                        if (keys['KeyW'] || keys['ArrowUp']) {
                            if (v.gear > 0 && v.speed < gearCfg[v.gear].max + 0.05) acc = gearCfg[v.gear].acc;
                            else if (v.gear === -1 && v.speed > gearCfg[v.gear].max) acc = gearCfg[v.gear].acc;
                        }
                        // S (freio): fora do drift desacelera normalmente; dentro do drift, desacelera sem sair do drift
                        // O tratamento do S no drift é feito mais abaixo (antes do drag)
                        
                        const turn = (keys['KeyA'] || keys['ArrowLeft']) ? 0.05 : (keys['KeyD'] || keys['ArrowRight']) ? -0.05 : 0;
                        
                        let isDriftMode = false;
                        let isBurnout = false;
                        if (v.gear === 1) {
                            if (spacePressed) {
                                isBurnout = true; v.speed *= 0.92;
                                if (Math.random() > 0.3) { createEffect(v.mesh.position, 0xffffff, 0.4, 1.2); createSkid(v.mesh.position, v.carAngle); }
                            }
                        } else if (v.gear === 2) {
                            if (spacePressedThisFrame) driftModeActive[idx] = !driftModeActive[idx];
                            isDriftMode = driftModeActive[idx];
                        } else if (v.gear >= 3) {
                            if (spacePressed) isDriftMode = true;
                        }
                        
                        spacePressedThisFrame = false;
                        if (Math.abs(v.speed) < 0.1) { isDriftMode = false; driftModeActive[idx] = false; v.driftIntensity = 0; }
                        
                        // === DRIFT: O CARRO REALMENTE VAI DE LADO CONFORME O ÂNGULO ===
                        // carAngle = onde o carro APONTA (visual)
                        // moveAngle = para onde o carro VAI (movimento)
                        // No drift, o carro aponta ~70° de lado do movimento, E vai de lado na mesma proporção
                        // Isso elimina o efeito "fantasma" — o visual bate com o movimento
                        
                        let rawSlip = v.carAngle - v.moveAngle;
                        while (rawSlip > Math.PI) rawSlip -= 2 * Math.PI;
                        while (rawSlip < -Math.PI) rawSlip += 2 * Math.PI;
                        v.slipAngle = rawSlip;
                        
                        // Se pressionar S no drift, freia mas NÃO sai do drift
                        if (isDriftMode && (keys['KeyS'] || keys['ArrowDown'])) {
                            // Desacelera suavemente mantendo o drift ativo
                            v.speed = Math.max(0.05, v.speed - 0.006);
                            // NÃO aplica o freio pesado normal (acc) pra não sair do drift
                            let drag = 1.0 - v.driftIntensity * 0.012;
                            v.speed = v.speed * drag;
                        } else if (keys['KeyS'] || keys['ArrowDown']) {
                            // Fora do drift: freio normal
                            const brakePower = Math.abs(v.speed) > 0.5 ? -0.02 : -0.04;
                            acc = v.speed > 0 ? brakePower : -brakePower;
                            let drag = 0.997;
                            v.speed = Math.max(-0.2, Math.min(v.speed + acc, 1.0)) * drag;
                        } else {
                            let drag = isDriftMode ? (1.0 - v.driftIntensity * 0.012) : 0.997;
                            v.speed = Math.max(-0.2, Math.min(v.speed + acc, 1.0)) * drag;
                        }

                        const driftIntensityTarget = isDriftMode ? 1.0 : 0.0;
                        v.driftIntensity += (driftIntensityTarget - v.driftIntensity) * 0.10;

                        const maxDriftAngle = 70 * Math.PI / 180; // 70 graus

                        if (Math.abs(v.speed) > 0.04) {
                            // A direção controla o movimento (carAngle) — onde o carro vai
                            // No drift, o carro vai pra onde você aponta
                            const speedFactor = Math.max(0.6, 1.0 - Math.abs(v.speed) * 0.3);
                            const turnForce = turn * speedFactor * 1.2;
                            v.carAngle += turnForce;

                            if (isDriftMode) {
                                // O moveAngle (movimento real) fica "preso" e muda mais lentamente
                                // Isso cria o efeito: carro aponta numa direção mas vai de lado
                                // Quanto mais drift, mais o movimento fica "preso" = mais inclinação visível
                                const gripLoss = v.driftIntensity * 0.55; // 55% do movimento fica preso
                                v.moveAngle += (v.carAngle - v.moveAngle) * (1.0 - gripLoss) * 0.08;
                                
                                // Adiciona momentum lateral — o carro continua indo pro lado que estava
                                // Isso é o que faz ele realmente ir de lado
                                const lateralMomentum = v.slipAngle * 0.025 * v.driftIntensity;
                                v.moveAngle -= lateralMomentum;
                            } else {
                                // Fora do drift: moveAngle segue carAngle rapidamente (recupera grip)
                                v.moveAngle += (v.carAngle - v.moveAngle) * 0.15;
                            }
                        }

                        // Se soltou o drift, moveAngle e carAngle voltam ao mesmo
                        if (!isDriftMode && v.driftIntensity > 0.05) {
                            v.moveAngle += (v.carAngle - v.moveAngle) * 0.10;
                        }

                        // Detectar drift: quando há diferença real entre aponta e vai
                        const driftAngleDeg = Math.abs(v.carAngle - v.moveAngle) * (180 / Math.PI);
                        v.isDrifting = v.driftIntensity > 0.3 && Math.abs(v.speed) > 0.15 && driftAngleDeg > 10;
                        
                        // Efeito visual de drift: rastro de neon + faíscas no swap
                        if (v.isDrifting && Math.random() > 0.4) {
                            const neonColor = v.data.neon;
                            for (let side of [-0.8, 0.8]) {
                                const sparkPos = v.mesh.position.clone();
                                sparkPos.x += Math.cos(v.moveAngle) * side;
                                sparkPos.z += -Math.sin(v.moveAngle) * side;
                                sparkPos.y = 0.1;
                                createEffect(sparkPos, neonColor, 0.25 + v.driftIntensity * 0.3, 0.8 + v.driftIntensity * 0.5);
                            }
                        }
                        // Faíscas extras durante swap de direção
                        if (isDriftMode && turn !== 0 && Math.abs(turn) > 0.01) {
                            if (Math.random() > 0.6) {
                                const sparkPos = v.mesh.position.clone();
                                sparkPos.y = 0.15;
                                createEffect(sparkPos, 0xffaa00, 0.3, 1.5);
                            }
                        }

                        const nx = v.mesh.position.x + Math.sin(v.moveAngle) * v.speed;
                        const nz = v.mesh.position.z + Math.cos(v.moveAngle) * v.speed;
                        
                        let hit = false;
                        obstacles.forEach(o => { if (new THREE.Vector2(nx, nz).distanceTo(new THREE.Vector2(o.pos.x, o.pos.z)) < v.radius + o.radius) hit = true; });
                        vehicles.forEach((other, oi) => { if (oi !== idx && v.mesh.position.distanceTo(other.mesh.position) < v.radius + other.radius) hit = true; });

                        if (hit) {
                            v.speed = -v.speed * 0.5; v.driftPoints = 0; multiplier = 1.0;
                            document.getElementById('crash-msg').style.display = 'block';
                            setTimeout(() => document.getElementById('crash-msg').style.display = 'none', 500);
                        } else {
                            v.mesh.position.x = nx; v.mesh.position.z = nz;
                        }
                        v.mesh.rotation.y = v.carAngle;

                        if (v.isDrifting) {
                            v.driftPoints += 25; multiplier = Math.min(10, multiplier + 0.008);
                            // Mais partículas e skids no drift intenso
                            const skidChance = 0.3 + v.driftIntensity * 0.4;
                            const smokeCount = Math.floor(v.driftIntensity * 2);
                            for (let s = 0; s < smokeCount; s++) {
                                if (Math.random() > skidChance) createEffect(v.mesh.position, 0xdddddd, 0.3 + v.driftIntensity * 0.3, 1.0 + v.driftIntensity * 0.5);
                            }
                            if (Math.random() > skidChance) createSkid(v.mesh.position, v.carAngle);
                        } else if (!v.isDrifting && v.driftPoints > 0) {
                            totalMoney += Math.floor(v.driftPoints * multiplier); v.driftPoints = 0; multiplier = 1.0;
                        }

                        document.getElementById('drift-points').innerText = v.driftPoints;
                        document.getElementById('speed-val').innerText = Math.round(v.speed * 120);
                        document.getElementById('car-name').innerText = vehicles[activeIdx].data.name;
                        document.getElementById('gear-val').innerText = v.gear === -1 ? 'R' : v.gear === 0 ? 'N' : v.gear;
                        
                        // Indicador visual de drift
                        const driftIndicator = document.getElementById('drift-indicator');
                        const body = document.body;
                        if (v.isDrifting) {
                            driftIndicator.style.display = 'block';
                            driftIndicator.innerText = `DRIFT! x${multiplier.toFixed(1)}`;
                            body.classList.add('drift-active');
                        } else {
                            driftIndicator.style.display = 'none';
                            body.classList.remove('drift-active');
                        }
                        
                        // Câmera SUAVE: SEM TRAVADAS, SEM PULOS
                        // camAngle, camDist, camHeight são interpolados suavemente
                        // Inicializa no primeiro frame na posição atual do carro
                        if (!cameraInitialized) {
                            camAngle = v.carAngle;
                            camDist = 12 + v.speed * 8;
                            camHeight = 6;
                            cameraInitialized = true;
                        }
                        
                        // Suavização angular: normaliza diferença e interpola
                        let angleDiff = v.carAngle - camAngle;
                        while (angleDiff > Math.PI) angleDiff -= 2 * Math.PI;
                        while (angleDiff < -Math.PI) angleDiff += 2 * Math.PI;
                        
                        // Interpolação angular suave (não trava, não pula)
                        const angularLerp = 0.045;
                        camAngle += angleDiff * angularLerp;
                        
                        // Suavização da distância da câmera (evita pulo ao ativar/soltar drift)
                        let targetDist = 12 + v.speed * 8;
                        if (v.isDrifting && v.driftIntensity > 0.3) {
                            targetDist -= 2.5 * v.driftIntensity;
                        }
                        const distLerp = 0.06;
                        camDist += (targetDist - camDist) * distLerp;
                        
                        // Suavização da altura da câmera
                        let targetHeight = 6;
                        if (v.isDrifting && v.driftIntensity > 0.3) {
                            targetHeight -= 1.2 * v.driftIntensity;
                        }
                        const heightLerp = 0.06;
                        camHeight += (targetHeight - camHeight) * heightLerp;
                        
                        // FOV suave também
                        let targetFov = 60 + v.speed * 15;
                        if (v.isDrifting && v.driftIntensity > 0.5) {
                            targetFov += 5 * v.driftIntensity;
                        }
                        const fovLerp = 0.06;
                        camera.fov += (targetFov - camera.fov) * fovLerp;
                        camera.updateProjectionMatrix();
                        
                        camera.position.set(
                            v.mesh.position.x - Math.sin(camAngle) * camDist,
                            camHeight,
                            v.mesh.position.z - Math.cos(camAngle) * camDist
                        );
                        camera.lookAt(v.mesh.position.x, v.mesh.position.y + 1, v.mesh.position.z);

                    } else {
                        // === NPC COM DRIFT REALISTA: SEGUE TRAÇADO + SPAN DE LADO A LADO ===
                        
                        if (!v.npcTarget) v.npcTarget = 0.001 + idx * 0.15;
                        if (v.npcTarget > 1) v.npcTarget = v.npcTarget - 1;
                        
                        // Ponto-alvo na pista (sempre um pouco à frente)
                        const lookAhead = 0.008;
                        const targetPt = trackCurve.getPointAt(v.npcTarget);
                        const lookPt = trackCurve.getPointAt((v.npcTarget + lookAhead) % 1);
                        
                        // Direção da pista no ponto-alvo
                        const dir = new THREE.Vector3().subVectors(lookPt, targetPt).normalize();
                        const targetAngle = Math.atan2(dir.x, dir.z);
                        
                        // Detectar curvas: diferença entre direção atual e próxima
                        const curDir = new THREE.Vector3(Math.sin(targetAngle), 0, Math.cos(targetAngle));
                        const oldDir = new THREE.Vector3(Math.sin(v._npcLastAngle || targetAngle), 0, Math.cos(v._npcLastAngle || targetAngle));
                        v._npcLastAngle = targetAngle;
                        const curveSharpness = 1 - curDir.dot(oldDir); // 0 = reto, maior = mais curvo
                        
                        // --- VELOCIDADE: acelera nas retas, desacelera nas curvas ---
                        let targetSpeed = 0.32;
                        if (curveSharpness > 0.25) targetSpeed = 0.14;  // Curva muito fechada
                        else if (curveSharpness > 0.1) targetSpeed = 0.22; // Curva média
                        
                        if (v.speed < targetSpeed) v.speed += 0.003;
                        else v.speed -= 0.0015;
                        v.speed = Math.max(0.06, Math.min(v.speed, 0.45));
                        
                        // --- SPAN DE LADO A LADO: o NPC oscila entre os lados da pista ---
                        // Gera um offset lateral que oscila suavemente
                        if (v._npcSpanTime === undefined) {
                            v._npcSpanTime = Math.random() * 100;
                            v._npcSpanPhase = Math.random() < 0.5 ? 1 : -1;
                            v._npcSpanSpeed = 0.008 + Math.random() * 0.005;
                        }
                        v._npcSpanTime += v._npcSpanSpeed;
                        const spanOscillation = Math.sin(v._npcSpanTime) * 0.5;
                        const spanTarget = spanOscillation * 10; // ±10 unidades de lado (pista larga)
                        
                        // Calcula o lado direito da pista
                        const sideDir = new THREE.Vector3(-dir.z, 0, dir.x);
                        
                        // Posição desejada: centro da pista + offset lateral (span)
                        let desiredX = targetPt.x + sideDir.x * spanTarget;
                        let desiredZ = targetPt.z + sideDir.z * spanTarget;
                        
                        // CarAngle aponta para a posição desejada
                        let angleToDesired = Math.atan2(desiredX - v.mesh.position.x, desiredZ - v.mesh.position.z);
                        let angleDiff = angleToDesired - v.carAngle;
                        while (angleDiff > Math.PI) angleDiff -= 2 * Math.PI;
                        while (angleDiff < -Math.PI) angleDiff += 2 * Math.PI;
                        v.carAngle += angleDiff * 0.05;
                        
                        // --- DRIFT REAL: moveAngle com inércia lateral ---
                        const isDriftMode = curveSharpness > 0.08 && v.speed > 0.1;
                        v.driftIntensity += ((isDriftMode ? 1.0 : 0.0) - v.driftIntensity) * 0.05;
                        
                        if (isDriftMode && v.driftIntensity > 0.2) {
                            let slipDiff = v.carAngle - v.moveAngle;
                            while (slipDiff > Math.PI) slipDiff -= 2 * Math.PI;
                            while (slipDiff < -Math.PI) slipDiff += 2 * Math.PI;
                            v.moveAngle += slipDiff * (1.0 - v.driftIntensity * 0.45) * 0.04;
                            const latMom = slipDiff * 0.018 * v.driftIntensity;
                            v.moveAngle -= latMom;
                        } else {
                            if (v.moveAngle === undefined) v.moveAngle = v.carAngle;
                            v.moveAngle += (v.carAngle - v.moveAngle) * 0.08;
                        }
                        
                        // Movimentação usando moveAngle (com inércia de drift)
                        const nx = v.mesh.position.x + Math.sin(v.moveAngle) * v.speed;
                        const nz = v.mesh.position.z + Math.cos(v.moveAngle) * v.speed;
                        
                        // Verifica se está na pista
                        let onTrack = false;
                        for(let i = 0; i < trackCheckPoints.length; i += 2) {
                            if (new THREE.Vector2(nx, nz).distanceTo(new THREE.Vector2(trackCheckPoints[i].x, trackCheckPoints[i].z)) < 22) {
                                onTrack = true; break;
                            }
                        }
                        
                        if (onTrack) {
                            v.mesh.position.x = nx;
                            v.mesh.position.z = nz;
                        } else {
                            // Volta pra o centro da pista
                            const p = trackCurve.getPointAt(v.npcTarget);
                            v.mesh.position.x = p.x;
                            v.mesh.position.z = p.z;
                        }
                        
                        v.mesh.rotation.y = v.carAngle;
                        
                        // Avança na pista (baseado na velocidade)
                        v.npcTarget += (v.speed * 0.0025);
                        if (v.npcTarget > 1) v.npcTarget = v.npcTarget - 1;
                        
                        // === EFEITOS VISUAIS DE DRIFT ===
                        if (v.driftIntensity > 0.25 && Math.random() > 0.45) {
                            createEffect(v.mesh.position.clone(), v.data.neon, 0.15 + v.driftIntensity * 0.15, 1.0);
                        }
                        if (v.driftIntensity > 0.3 && Math.random() > 0.5) {
                            createSkid(v.mesh.position.clone(), v.carAngle);
                        }
                        // Faíscas de drift
                        if (v.driftIntensity > 0.4 && Math.random() > 0.7) {
                            createEffect(v.mesh.position.clone(), 0xffaa00, 0.2, 1.5);
                        }
                    }
                });

                document.getElementById('multiplier').innerText = `x${multiplier.toFixed(1)}`;
                document.getElementById('money').innerText = `$ ${totalMoney}`;

                for(let i=particles.length-1; i>=0; i--) {
                    particles[i].life -= particles[i].decay;
                    particles[i].mesh.material.opacity = particles[i].life * 0.4;
                    particles[i].mesh.scale.multiplyScalar(particles[i].grow);
                    if (particles[i].life <= 0) { scene.remove(particles[i].mesh); particles.splice(i, 1); }
                }
                for(let i=skids.length-1; i>=0; i--) {
                    skids[i].life -= 0.005; skids[i].mesh.material.opacity = skids[i].life * 0.2;
                    if (skids[i].life <= 0) { scene.remove(skids[i].mesh); skids.splice(i, 1); }
                }
                renderer.render(scene, camera);
            }
            animate();
            window.onresize = () => { camera.aspect = window.innerWidth / window.innerHeight; camera.updateProjectionMatrix(); renderer.setSize(window.innerWidth, window.innerHeight); };
        };
    </script>
</body>
